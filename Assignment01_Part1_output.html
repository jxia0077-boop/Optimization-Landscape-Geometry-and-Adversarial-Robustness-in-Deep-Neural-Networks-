(g) Why is the cross-entropy loss caused by the feed-forward neural network at $(x,y)$ (i.e., $\text{CE}(1_y, p(x))$) always non-negative? When does this $\text{CE}(1_y, p(x))$ loss get the value $0$? Note that you need to answer this question for a general pair $(x,y)$ and a general feed-forward neural network with, for example $M=4$ classes?

[1 point]
Answer: For $M$ classes, $\mathrm{CE}(1_y,p)=-\sum_{i=1}^M \mathbf{1}\{i=y\}\log p_i=-\log p_y\ge 0 $ because $0\le p_y\le1\Rightarrow \log p_y\le0$. Equivalently, $\mathrm{CE}(1_y,p)=\mathrm{KL}(1_y\Vert p)\ge0$. The loss equals $\boxed{0}$ iff the model assigns probability $1$ to the true class, i.e., $p_y=1$ (so $p=1_y$).

class MyFFN(torch.nn.Module):
  def __init__(self, input_size, num_classes, hidden_sizes, act = torch.nn.ReLU()):
    """
    input_size: the size of the input
    num_classes: the number of classes
    act is the activation function
    hidden_sizes is the list of hidden sizes
    for example input_size = 3, hidden_sizes = [5, 7], num_classes = 4, and act = torch.nn.ReLU()
    means that we are building up a FFN with the confirguration
    (3 (Input) -> 5 (ReLU) -> 7 (ReLU) -> 4 (Output))
    """
    super(MyFFN, self).__init__()
    self.input_size = input_size
    self.num_classes = num_classes
    self.hidden_sizes = hidden_sizes
    self.num_layers = len(hidden_sizes) + 1


    if isinstance(act, torch.nn.Module):
      self._act_factory = lambda: act
    else:

      self._act_factory = act

    self.layers = torch.nn.ModuleList()
    self.acts = torch.nn.ModuleList()


    self._momentum_bufs = None
    self._adagrad_sums = None


  def create_FFN(self):
    """
    This function creates the feed-forward neural network
    We stack many MyLinear layers
    """
    sizes = [self.input_size] + self.hidden_sizes + [self.num_classes]
    #Your code here
    self.layers = torch.nn.ModuleList()
    self.acts = torch.nn.ModuleList()
    for i in range(len(sizes) - 2):
      self.layers.append(MyLinear(sizes[i], sizes[i+1]))
      self.acts.append(self._act_factory() if callable(self._act_factory) else self._act_factory)

    self.layers.append(MyLinear(sizes[-2], sizes[-1]))


    self.to(device)

  def forward(self, x):
    """
    This implements the forward propagation of the batch x
    This needs to return the logits of x
    """
    #Your code here
    h = x.to(device)

    for i in range(len(self.layers) - 1):
      h = self.layers[i](h)
      h = self.acts[i](h)
    logits = self.layers[-1](h)
    return logits

  def compute_loss(self, x, y):
    """
    This function computes the cross-entropy loss
    You can use the built-in CE loss of PyTorch
    """
    #Your code here
    logits = self.forward(x)
    y = y.to(device)
    loss = F.cross_entropy(logits, y, reduction='mean')
    return loss

  def _ensure_momentum_bufs(self):
    if self._momentum_bufs is None:
      self._momentum_bufs = [torch.zeros_like(p, device=p.device) for p in self.parameters()]

  def _ensure_adagrad_sums(self):
    if self._adagrad_sums is None:
      self._adagrad_sums = [torch.zeros_like(p, device=p.device) for p in self.parameters()]

  def update_SGD(self, x, y, learning_rate = 0.01):
    """
    This function updates the model parameters using SGD using the batch (x,y)
    You need to implement the update rule manually and cannot rely on the built-in optimizer
    """
    #Your code here
    self.train()
    self.zero_grad()
    loss = self.compute_loss(x, y)
    loss.backward()

    with torch.no_grad():
      for p in self.parameters():
        if p.grad is None: continue
        p -= learning_rate * p.grad
    self.zero_grad()
    return float(loss.detach())

  def update_SGDwithMomentum(self, x, y, learning_rate = 0.01, momentum = 0.9):
    """
    This function updates the model parameters using SGD with momentum using the batch (x,y)
    You need to implement the update rule manually and cannot rely on the built-in optimizer
    """
    #Your code here
    self.train()
    self.zero_grad()
    loss = self.compute_loss(x, y)
    loss.backward()
    self._ensure_momentum_bufs()

    with torch.no_grad():

        if p.grad is None: continue
        buf.mul_(momentum).add_(p.grad)
        p.add_(-learning_rate * buf)
    self.zero_grad()
    return float(loss.detach())

  def update_AdaGrad(self, x, y, learning_rate = 0.01, eps=1e-8):
    """
    This function updates the model parameters using AdaGrad using the batch (x,y)
    You need to implement the update rule manually and cannot rely on the built-in optimizer

    self.zero_grad()
    loss = self.compute_loss(x, y)
    loss.backward()

        G.add_(p.grad * p.grad)

def compute_acc(model, data_loader):
  """
  This function computes the accuracy of the model on a data loader
  """
  #Your code here
  model.eval()
  correct, total = 0, 0
  with torch.no_grad():
    for xb, yb in data_loader:
      xb = xb.to(device)
      yb = yb.to(device)
      logits = model(xb)
      preds = logits.argmax(dim=1)
      correct += (preds == yb).sum().item()
      total += yb.numel()
  return correct / total

def compute_loss(model, data_loader):
  """
  This function computes the loss of the model on a data loader
  """
  #Your code here
  model.eval()
  total_loss, total = 0.0, 0
  with torch.no_grad():
    for xb, yb in data_loader:
      xb = xb.to(device)
      yb = yb.to(device)
      logits = model(xb)
      loss = F.cross_entropy(logits, yb, reduction='sum')
      total_loss += loss.item()
      total += yb.numel()
  return total_loss / total

def train_with_updater(model, train_loader, test_loader, num_epochs=50, lr=0.01, updater="sgd", momentum=0.9):
  """
  updater: "sgd" | "momentum" | "adagrad"
  """
  for epoch in range(num_epochs):
    for xb, yb in train_loader:
      if updater == "sgd":
        model.update_SGD(xb, yb, learning_rate=lr)
      elif updater == "momentum":
        model.update_SGDwithMomentum(xb, yb, learning_rate=lr, momentum=momentum)
      elif updater == "adagrad":
        model.update_AdaGrad(xb, yb, learning_rate=lr)
      else:
        raise ValueError("Unknown updater")
    train_acc = compute_acc(model, train_loader)
    train_loss = compute_loss(model, train_loader)
    test_acc = compute_acc(model, test_loader)
    test_loss = compute_loss(model, test_loader)
    print(f"[{updater.upper()}] Epoch {epoch+1:02d}/{num_epochs} | "
       f"Train Loss: {train_loss:.4f} | Train Acc: {train_acc*100:.2f}% | "
       f"Test Loss: {test_loss:.4f} | Test Acc: {test_acc*100:.2f}%")
